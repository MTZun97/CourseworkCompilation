### PART 1:  Analaytical survival functions

# Define ranges of thresholds
u0   = np.linspace(-1, 2, 1001)  # random variable range for the uniform distribution
u1   = np.linspace(-4, 4, 1001)  # random variable range for the normal distribution
u2   = np.linspace(0, 20, 1001)  # random variable range for the chi2 distribution


# Use analytical equations to calculate the probability densities at all x values 
sf0  = stats.uniform.sf(u0)     # SF for a uniform random variable
sf1  = stats.norm.sf(u1)        # SF for a normal random variable
sf2  = stats.chi2.sf(u2, 7)     # SF for a chi2 random variable (with degrees of freedom = 7)



### PART 2:  Numerical survival functions

# generate random values:
x0   = np.random.rand(10000)         # uniform
x1   = np.random.randn(10000)        # normal
x2   = np.random.chisquare(7, 10000) # chi-square with 7 degrees of freedom


# specify thresholds for numerical survival function calculations:
u0n  = np.linspace(-1, 2, 51)
u1n  = np.linspace(-4, 4, 51)
u2n  = np.linspace(0, 20, 51)


# calculate survival functions numerically:
sf0n = [(x0>uu).mean()  for uu in u0n]
sf1n = [(x1>uu).mean()  for uu in u1n]
sf2n = [(x2>uu).mean()  for uu in u2n]



### PART 3:  Plot

fig,ax = plt.subplots(1, 3, figsize=(14,3))

ax[0].plot(u0, sf0, linewidth=2, zorder=1, label='Analytical')
ax[1].plot(u1, sf1, linewidth=2, zorder=1)
ax[2].plot(u2, sf2, linewidth=2, zorder=1)

ax[0].plot(u0n, sf0n, 'o', zorder=0, label='Numerical')
ax[1].plot(u1n, sf1n, 'o', zorder=0)
ax[2].plot(u2n, sf2n, 'o', zorder=0)

ax[0].legend()

ax[0].set_title('Uniform')
ax[1].set_title('Normal')
ax[2].set_title('$\chi^2$  ($v$ = 7)')

plt.setp(ax, xlabel='u')
plt.setp(ax, ylabel='Survival function')

[axx.axhline(0, color='k', linestyle=':')  for axx in ax]
[axx.axhline(1, color='k', linestyle=':')  for axx in ax]

plt.show()

np.random.seed(0)
n      = 10
niter  = 10000
x      = np.arange(n)
r      = []
for i in range(niter):
    y  = np.random.randn(n)
    r.append( np.corrcoef(x, y)[0,1] )
r      = np.array(r)

p      = (r > 0.5).mean()

print('When there is truly no correlation between two variables,')
print('  the probability that the correlation coefficient exceeds 0.5 is:')
print('  P( r > 0.5) = ', p)
print()

plt.figure()
ax = plt.axes()
ax.hist(r, bins=21)
ax.axvline(0.5, color=(1,0,0), linestyle='--')
plt.show()
